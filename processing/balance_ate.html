db <- db %>% 
  rename(Aleatorio = aleatorio)

results_list <- list()

  formula <- as.formula(paste(dep_var, "~ Aleatorio + religion"))
  result <- lm_robust(formula = formula, data = db, se_type = "HC2", alpha = 0.05)
  results_list[[dep_var]] <- result
}
